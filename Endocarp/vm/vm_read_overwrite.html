vm_read_overwrite()

MIT Documentation for vm_read_overwrite()

Example

/*
read out from task's address space
*/

vm_address_t address = 0x10086c000;
uint8_t read_buffer[4];
vm_size_t size = 4;

kern_return_t kret;

kret = vm_read_overwrite(mach_task_self(), //task for reading in
               address, //address where read begins
               size, //in size of read
               (vm_offset_t) read_buffer, //where data will be read to
               &size); // out size

if (kret != KERN_SUCCESS) {
     return kret;
}
return KERN_SUCCESS;
