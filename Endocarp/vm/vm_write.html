vm_write()

MIT Documentation for vm_write()

Example

/*
write to task's address space
*/

vm_address_t address = 0x10086c000;
uint8_t data[] = {0x00, 0x00, 0x00, 0x00};
vm_size_t size = 4;

kern_return_t kret;

kret = vm_write(mach_task_self(), //task for to write data
               address, //address to write data at
               (vm_offset_t) &data, //bytes for writing
               size); //size of data

if (kret != KERN_SUCCESS) {
     return kret;
}
return KERN_SUCCESS;
