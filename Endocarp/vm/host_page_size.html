host_page_size()

MIT Documentation for host_page_size()

Example

/*
return host page size
will return 16k (0x4000) >= iOS 9
*/

vm_size_t size = 0;
kern_return_t kret;

kret = host_page_size(mach_host_self(), &size);
printf("Host Page Size: 0x%x\n", size);

if (kret != KERN_SUCCESS) {
     return kret;
}
return KERN_SUCCESS;

Notes

On iOS 9 and above host_page_size will return 16,384b (0x4000).
