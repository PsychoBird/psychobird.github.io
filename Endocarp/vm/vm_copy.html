vm_copy()

MIT Documentation for vm_copy()

Example

/*
copy data from address to another address in the task memory
must copy data on page boundaries
*/

vm_address_t copy_at = 0x10086c000;
vm_size_t count = 0x4000;
vm_address_t copy_to = 0x10087e000;

kern_return_t kret;

kret = vm_copy(mach_task_self(), // send right to task to copy in
               copy_at, // source address to copying
               count, // amount of bytes to copy
               copy_to); // copy destination

if (kret != KERN_SUCCESS) {
     return kret;
}
return KERN_SUCCESS;
